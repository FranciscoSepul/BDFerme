CREATE OR REPLACE PROCEDURE Todos_Clientes(P_RUNCLIENTE IN VARCHAR2) AS
Begin
SELECT iddireccion,runcliente,digitoverif,estado,nombre,apellido,sexo,correo,contrasena,region,comuna,pasaje,numero,numerodepto from Cliente
INNER JOIN DIRECCION ON direccion.id = cliente.iddireccion
where runCliente=P_RUNCLIENTE;
End;



EXECUTE Todos_Clientes(12423531);
